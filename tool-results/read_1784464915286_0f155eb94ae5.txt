  1232→function getFilteredProducts(){
  1233→  let list = [...API.products.getAll()];
  1234→  const f = state.filters;
  1235→  if(f.search) list = list.filter(p => (p.name+p.desc).toLowerCase().includes(f.search.toLowerCase()));
  1236→  if(f.category!=='all') list = list.filter(p=>p.category===f.category);
  1237→  if(f.country!=='all') list = list.filter(p=>p.country===f.country);
  1238→  if(f.price!=='all'){ const [lo,hi] = f.price.split('-').map(Number); list = list.filter(p=>p.price>=lo && p.price<=hi); }
  1239→  if(f.availability!=='all') list = list.filter(p=> stockStatus(p).key===f.availability);
  1240→  switch(f.sort){
  1241→    case 'price-asc': list.sort((a,b)=>a.price-b.price); break;
  1242→    case 'price-desc': list.sort((a,b)=>b.price-a.price); break;
  1243→    case 'newest': list.sort((a,b)=>a.addedDaysAgo-b.addedDaysAgo); break;
  1244→    case 'stock': list.sort((a,b)=>b.stock-a.stock); break;
  1245→    case 'alpha': list.sort((a,b)=>a.name.localeCompare(b.name)); break;
  1246→  }
  1247→  return list;
  1248→}
